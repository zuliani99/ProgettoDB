posts = Post.query.paginate()
dir(posts) -> metodi dell'oggetto
posts.per_page -> di default è 20
for post in posts.items
	print(post) -> ci stampa i primi 20

posts = Post.query.paginate(page=2)
for post in posts.items
	print(post) -> ci stampa glia altri 5 post

posts = Post.query.paginate(per_page=5) -> una pagina avrà 5 post
for post in posts.items
	print(post) -> ci stampa i primi 5 post

posts = Post.query.paginate(per_page=5, page=2)
for post in posts.items
	print(post) -> ci stampa i post dal 6 al 10
